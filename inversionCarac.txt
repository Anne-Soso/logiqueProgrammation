début inversionCarac
    déclaration Variables locales
        var texteSrc :T <- ""
        var newText :T <- ""
        var srcLen :N <- 0
        var cpt :N <- 0
    fin Variables locales

    Afficher "Saisir un texte à inverser"
    saisir texteSrc

    pour cpt allant de 1 à srcLen
        newText <- caract(texteSrc,cpt) (+) newText
    cpt suivant

    Afficher newText
    
début inversionCarac
