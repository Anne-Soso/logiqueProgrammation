début supprimerEspaces
    déclaration Variables Locales
        const TAILLE:N <- ...
        var tab:N[TAILLE]
        var ind:N <- 0
        var cptEs:N <- 0
    fin Variables Locales

    /****SAISIE CACHEE****/

    pour ind allant de 1 à TAILLE
        si tab[ind] != " "
            ALORS
                tab[ind-cptEs] <- tab[ind]
            SINON
                cptEs <- cptEs+1
        finsi
    ind suivant

    pour ind allant de TAILLE à TAILLE-cpt par pas de -1
        tab[ind] <- " "
    ind suivant
fin supprimerEspaces
